def createRandomEnvironment( height, width, n_obstacles, n_goals ):
  obstacles = []
  while (len(obstacles) < n_obstacles):
    ob = ( random.randint(0,height-1), random.randint(0,width-1) )
    if ( ob not in obstacles ) and ( ob != (0,0) ):
      obstacles.append( ob )

  goal_coord = []
  goal = []

  while( len(goal_coord) < n_goals ):
    rc = ( random.randint(0,height-1), random.randint(0,width-1) )
    r = 1 #random.randint(-99,99)
    if ( rc not in goal_coord ) and ( rc not in obstacles ):
      goal_coord.append( rc )
      goal.append(r) 
  
  return (height, width, obstacles, [ x for x in zip(goal_coord, goal) ] )

    bg = mpatches.Rectangle( (ox,oy), 1.0, 1.0, linewidth=2, color="#2020a0", fill=True )
    r = mpatches.Rectangle( (x,y), 1.0, 1.0, linewidth=2, color="black", fill=False )
    frame.append( ax.add_patch(bg) )

  for gc, g in goals:

def createGoalOnRight( env ):
  height, width, obstacles, _ = env
  goal = None
  gx = width - 1 
  while ( goal is None ):
    gy = random.randint(0,height-1)
    if ( (gy,gx) not in obstacles ):
      goal = (gy,gx)
    goals = [ ( goal, 1 ) ]
  return (height, width, obstacles, goals )

Robot Navigation Domain

  • Robot will start at location (A,0)
  • Goal (red square) is randomly placed on the right size
  • Robot can move up, down, left, and right
  • Blue squares are obstacles

def rn_is_valid( state ):
  ret = True

def rn_is_goal( state ):
  ret = False
  gc = [ ( g[0][0],g[0][1] ) for g in goals ]
  if state in gc:
    ret = True
  return ret

def rn_successors( state ):
  children = []
  cy,cx = state
  for dy, dx in [ (+1,0), (-1,0), (0,+1), (0,-1) ]:
    ny,nx = cy + dy, cx + dx
    if ( ny >= 0 ) and (ny < height) and (nx >= 0) and (nx < width):
      if rn_is_valid( (ny,nx) ):
        children.append( (ny, nx) )
  return children

Heuristic Function

  • A function h that measures how good (i.e. how close to the goal) a state is
  • Navigation domains usually use cost/distance functions
  • h=0 means we have reached the goal
  • h >> 0 means we are far away from the goal
  • Euclidean distance \( \sqrt{(x_r - x_g)^2 + (y_r - y_g)^2} \)
